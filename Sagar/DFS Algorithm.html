DFS Algorithm Code

#include <bits/stdc++.h>
using namespace std;

class Solution {
public:
    void dfs(int node, vector &vis, vector adj[], vector &result) {
        vis[node] = 1;
        result.push_back(node);

        for (auto it : adj[node]) {
            if (!vis[it]) {
                dfs(it, vis, adj, result);
            }
        }
    }

    vector dfsOfGraph(int V, vector adj[]) {
        vector result;
        vector vis(V, 0);
        dfs(0, vis, adj, result);
        return result;
    }
};

int main() {
    int V = 5;
    vector adj[V];

    adj[0] = {1, 2};
    adj[1] = {0, 3, 4};
    adj[2] = {0};
    adj[3] = {1};
    adj[4] = {1};

    Solution obj;
    vector result = obj.dfsOfGraph(V, adj);

    for (int i : result) {
        cout << i << " ";
    }

    return 0;
}
