// Function 1: Optimizing Network Infrastructure using Dijkstra's Algorithm
void optimizeNetworkInfrastructure() {
    int n, m;
    cout << "Enter number of nodes and edges: ";
    cin >> n >> m;

    vector> adj[n];
    cout << "Enter edges (u v w):\n";
    for (int i = 0; i < m; i++) {
        int u, v, w;
        cin >> u >> v >> w;
        adj[u].push_back({v, w});
        adj[v].push_back({u, w});
    }

    vector dist(n, INT_MAX);
    priority_queue, vector>, greater>> pq;

    int src;
    cout << "Enter source node: ";
    cin >> src;
    dist[src] = 0;
    pq.push({0, src});

    while (!pq.empty()) {
        int d = pq.top().first;
        int u = pq.top().second;
        pq.pop();

        if (d > dist[u]) continue;

        for (auto edge : adj[u]) {
            int v = edge.first;
            int weight = edge.second;

            if (dist[u] + weight < dist[v]) {
                dist[v] = dist[u] + weight;
                pq.push({dist[v], v});
            }
        }
    }

    cout << "Shortest distances from source node:\n";
    for (int i = 0; i < n; i++) {
        cout << "Node " << i << ": " << dist[i] << endl;
    }
}

// Function 2: Resource Allocation using AVL Tree (Simulated with a simple BST here)
struct Node {
    int key;
    Node *left, *right;
};

Node* createNode(int key) {
    Node* node = new Node;
    node->key = key;
    node->left = node->right = nullptr;
    return node;
}

Node* insert(Node* root, int key) {
    if (root == nullptr) return createNode(key);
    if (key < root->key)
        root->left = insert(root->left, key);
    else
        root->right = insert(root->right, key);
    return root;
}

void inorder(Node* root) {
    if (root == nullptr) return;
    inorder(root->left);
    cout << root->key << " ";
    inorder(root->right);
}

// Function 3: AI for Job Matching using KMP String Matching
void KMPSearch(string pattern, string text) {
    int m = pattern.size();
    int n = text.size();

    vector lps(m, 0);
    int j = 0;

    for (int i = 1; i < m; i++) {
        if (pattern[i] == pattern[j]) {
            j++;
            lps[i] = j;
        } else {
            if (j != 0) {
                j = lps[j - 1];
                i--;
            } else {
                lps[i] = 0;
            }
        }
    }

    int i = 0;
    j = 0;
    while (i < n) {
        if (pattern[j] == text[i]) {
            j++;
            i++;
        }

        if (j == m) {
            cout << "Pattern found at index " << i - j << endl;
            j = lps[j - 1];
        } else if (i < n && pattern[j] != text[i]) {
            if (j != 0) {
                j = lps[j - 1];
            } else {
                i++;
            }
        }
    }
}

// Function 4: Demand Prediction using Segment Tree
class SegmentTree {
    vector tree;
    vector arr;
    int n;

public:
    SegmentTree(vector& input) {
        n = input.size();
        arr = input;
        tree.resize(4 * n, 0);
        build(0, 0, n - 1);
    }

    void build(int node, int start, int end) {
        if (start == end) {
            tree[node] = arr[start];
        } else {
            int mid = (start + end) / 2;
            build(2 * node + 1, start, mid);
            build(2 * node + 2, mid + 1, end);
            tree[node] = tree[2 * node + 1] + tree[2 * node + 2];
        }
    }

    int query(int node, int start, int end, int l, int r) {
        if (r < start || l > end) return 0;
        if (l <= start && r >= end) return tree[node];

        int mid = (start + end) / 2;
        int leftSum = query(2 * node + 1, start, mid, l, r);
        int rightSum = query(2 * node + 2, mid + 1, end, l, r);
        return leftSum + rightSum;
    }

    int rangeQuery(int l, int r) {
        return query(0, 0, n - 1, l, r);
    }
};

// Function 5: Startup Collaboration using Kruskal's Algorithm
struct Edge {
    int u, v, weight;
};

bool compare(Edge a, Edge b) {
    return a.weight < b.weight;
}

int findParent(int u, vector& parent) {
    if (u == parent[u]) return u;
    return parent[u] = findParent(parent[u], parent);
}

void unionSet(int u, int v, vector& parent, vector& rank) {
    u = findParent(u, parent);
    v = findParent(v, parent);

    if (rank[u] < rank[v]) {
        parent[u] = v;
    } else if (rank[u] > rank[v]) {
        parent[v] = u;
    } else {
        parent[v] = u;
        rank[u]++;
    }
}

void startupCollaboration() {
    int n, m;
    cout << "Enter number of startups and connections: ";
    cin >> n >> m;

    vector edges(m);
    cout << "Enter connections (u v weight):\n";
    for (int i = 0; i < m; i++) {
        cin >> edges[i].u >> edges[i].v >> edges[i].weight;
    }

    sort(edges.begin(), edges.end(), compare);

    vector parent(n), rank(n, 0);
    for (int i = 0; i < n; i++) {
        parent[i] = i;
    }

    int cost = 0;
    cout << "Selected connections:\n";
    for (auto edge : edges) {
        if (findParent(edge.u, parent) != findParent(edge.v, parent)) {
            cost += edge.weight;
            cout << edge.u << " - " << edge.v << " : " << edge.weight << endl;
            unionSet(edge.u, edge.v, parent, rank);
        }
    }

    cout << "Total cost of collaboration: " << cost << endl;
}
